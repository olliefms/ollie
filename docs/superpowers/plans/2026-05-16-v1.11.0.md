async fn enrich_loads(
    state: &crate::AppState,
    items: Vec<crate::models::LoadListItem>,
) -> Vec<DispatchLoadListItem> {
    use std::collections::HashSet;
    let all_fac_ids: Vec<uuid::Uuid> = items.iter()
        .flat_map(|item| item.stops.iter().filter_map(|s| Some(s.facility_id)))
        .collect::<HashSet<_>>()
        .into_iter()
        .collect();
    let fac_map = state.db.batch_get_facilities(&all_fac_ids).await.unwrap_or_default();

    items.into_iter().map(|item| {
        let stops = item.stops.iter().map(|s| DispatchStopSummary {
            name: fac_map.get(&s.facility_id).map(|f| f.name.clone()),
            scheduled_arrive: s.scheduled_arrive.clone(),
        }).collect();
        DispatchLoadListItem {
            id: item.id,
            load_number: item.load_number,
            status: item.status.as_str().to_string(),
            customer_name: item.customer_name,
            stops,
            score: item.score,
        }
    }).collect()
}
```

    pub async fn batch_get_trailers(
        &self,
        ids: &[uuid::Uuid],
    ) -> Result<std::collections::HashMap<uuid::Uuid, crate::models::TrailerRecord>, AppError> {
        if ids.is_empty() { return Ok(std::collections::HashMap::new()); }
        let id_list = ids.iter().map(|id| format!("'{id}'")).collect::<Vec<_>>().join(", ");
        let stream = self.trailer_table.query()
            .only_if(format!("id IN ({id_list})"))
            .execute().await
            .map_err(|e| AppError::Internal(e.to_string()))?;
        Ok(batches_to_trailers(collect_stream(stream).await?)?
            .into_iter().map(|r| (r.id, r)).collect())
    }
```
