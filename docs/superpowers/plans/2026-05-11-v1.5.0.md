## Critical Constraints (read before touching anything)

- `merge_insert` only — never delete+insert in LanceDB
- `.only_if()` not `.filter()` for LanceDB conditional queries
- RFC3339 UTC+Z on all stored timestamps; IANA timezone for display conversion only
- Status transitions via action endpoints only, never PUT
- Append-only events table — never cascade-delete events
- Soft deletes before any hard delete
- All CSS must use design tokens from `base.css` — no inline hex values
- Never use `font-weight: 700+` for headings — 600 is the cap (DESIGN.md)
- Never use pill-shaped buttons — pills are for status badges only (DESIGN.md)
- LanceDB: use `.only_if()` not `.filter()` for conditional updates

- [ ] **Step A.3.1: Add base64urlToBuffer to auth.js**

In `static/driver/utils/auth.js`, add this export before the existing `bufferToBase64url` function:

```js
export function base64urlToBuffer(base64url) {
  const base64 = base64url.replace(/-/g, '+').replace(/_/g, '/');
  const padded = base64.padEnd(base64.length + (4 - base64.length % 4) % 4, '=');
  const binary = atob(padded);
  const bytes = new Uint8Array(binary.length);
  for (let i = 0; i < binary.length; i++) bytes[i] = binary.charCodeAt(i);
  return bytes.buffer;
}
```

      }
    } finally {
      setLoading(false);
    }
  });
```

export function formatStopType(type) {
  const labels = {
    'origin': 'ORIGIN',
    'fuel': 'FUEL',
    'pickup': 'PICKUP',
    'delivery': 'DELIVERY',
    'relay': 'RELAY',
    'empty_move': 'EMPTY MOVE',
    'maintenance': 'MAINTENANCE',
    'terminal': 'TERMINAL',
  };
  return labels[type] || type.toUpperCase();
}

export function formatWeight(lbs) {
  if (!lbs) return '0 lb';
  return lbs.toLocaleString() + ' lb';
}

export function formatStatus(status) {
  const labels = {
    'planned': 'Planned',
    'assigned': 'Assigned',
    'dispatched': 'Dispatched',
    'in_transit': 'In Transit',
    'delivered': 'Delivered',
    'completed': 'Completed',
    'cancelled': 'Cancelled',
  };
  return labels[status] || status;
}

// Format a UTC ISO string for display in a stop's local timezone.
// Falls back to browser locale if timezone is absent or unrecognized.
export function formatStopTime(dateStr, timezone) {
  if (!dateStr) return '—';
  const opts = {
    month: 'short', day: 'numeric',
    hour: '2-digit', minute: '2-digit',
  };
  if (timezone) {
    try {
      opts.timeZone = timezone;
      return new Date(dateStr).toLocaleString('en-US', opts);
    } catch {
      // fall through to locale default
    }
  }
  return new Date(dateStr).toLocaleString('en-US', opts);
}

export function formatShortTime(dateStr, timezone) {
  if (!dateStr) return '—';
  const opts = {
    month: '2-digit', day: '2-digit',
    hour: '2-digit', minute: '2-digit',
  };
  if (timezone) {
    try {
      opts.timeZone = timezone;
      return new Date(dateStr).toLocaleString('en-US', opts);
    } catch {
      // fall through
    }
  }
  return new Date(dateStr).toLocaleString('en-US', opts);
}
```
